package org.example; import com.sun.net.httpserver.HttpExchange; import com.sun.net.httpserver.HttpHandler; import org.apache.commons.lang3.StringEscapeUtils; import java.io.IOException; import java.io.OutputStream; public class ASHandler implements HttpHandler { @Override public void handle(HttpExchange he) throws IOException { String reVal; //httpResponse(he, reVal); } private void httpResponse(HttpExchange he, String reVal) throws IOException { OutputStream os = he.getResponseBody(); String hr = StringEscapeUtils.escapeHtml4(reVal); he.getResponseHeaders().add("Access-Control-Allow-Origin", "*"); he.getResponseHeaders().set("Content-Type", "text/html; charset=UTF-8"); he.sendResponseHeaders(200, hr.length()); os.write(hr.getBytes()); os.flush(); os.close(); } }
