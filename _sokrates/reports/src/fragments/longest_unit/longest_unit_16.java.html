    public static void StartServer(ThreadPoolExecutor tpe) {
        HttpServer hs;

        try {
            hs = HttpServer.create(new InetSocketAddress("localhost", 8001), 0);
            hs.createContext("/1/", new WHHandler());

            hs.setExecutor(tpe);
            hs.start();
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
